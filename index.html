üîß Method Examples

const clipper = new ImageClipper('#myImage');

// Change shape
clipper.setShape('heart');

// Set size (0-100)
clipper.setSize(75);

// Set position
clipper.setPosition(30, 70);  // x: 30%, y: 70%

// Set rotation
clipper.setRotation(45);      // 45 degrees

// Set multiple properties
clipper.set({
    shape: 'diamond',
    size: 60,
    x: 50,
    y: 50,
    rotation: 90
});

// Use presets
clipper.preset('heart');
clipper.preset('star');
clipper.preset('hexagon');

// Custom clip path
clipper.setCustomPath('polygon(20% 0%, 80% 0%, 100% 60%, 75% 100%, 25% 100%, 0% 60%)');

// Animate to new values
clipper.animate({
    shape: 'circle',
    size: 80,
    rotation: 360
}, 2000); // 2 seconds

// Get current values
const currentOptions = clipper.get();
const clipPath = clipper.getClipPath();

// Reset to defaults
clipper.reset();

// Remove clipping
clipper.clear();

üìö API Reference

Method Parameters Description
setShape(shape) string Set clipping shape: 'circle', 'ellipse', 'rectangle', 'triangle', 'diamond', 'hexagon', 'star', 'heart'
setSize(size) number (0-100) Set the size of the clipping area as percentage
setPosition(x, y) number, number Set X and Y position of the clipping area (0-100%)
setX(x) number (0-100) Set X position of the clipping area
setY(y) number (0-100) Set Y position of the clipping area
setRotation(degrees) number Set rotation in degrees (0-360)
set(options) object Set multiple options at once
preset(name) string Apply preset configuration
animate(options, duration) object, number Animate to new values over specified duration (ms)
get() - Get current options object
getClipPath() - Get current CSS clip-path value
reset() - Reset to default values
clear() - Remove all clipping effects
createControls(container) string|element Create interactive UI controls

üéØ Advanced Examples

// Clip multiple images at once
const clippers = ImageClipper.clipAll('.gallery img', {
    shape: 'hexagon',
    size: 80
});

// Chain methods
new ImageClipper('#hero-image')
    .setShape('circle')
    .setSize(90)
    .setPosition(60, 40);

// Event-driven clipping
document.getElementById('shape-select').addEventListener('change', (e) => {
    clipper.setShape(e.target.value);
});

// Responsive clipping
window.addEventListener('resize', () => {
    if (window.innerWidth < 768) {
        clipper.setSize(100);  // Full size on mobile
    } else {
        clipper.setSize(75);   // Clipped on desktop
    }
});

// Custom animations
clipper.animate({ rotation: 360 }, 1000)  // Spin
    .then(() => clipper.animate({ size: 20 }, 500))  // Shrink
    .then(() => clipper.animate({ size: 80 }, 500)); // Grow
